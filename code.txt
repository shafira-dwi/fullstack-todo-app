    <script>
        const form = document.getElementById('signin-form');

        form.addEventListener('submit', function (e) {
            e.preventDefault();

            const email = document.getElementById('email').value.trim();
            const password = document.getElementById('password').value.trim();

            const savedData = JSON.parse(localStorage.getItem('signupData')) || null;

            if (!savedData) {
                alert('Belum ada akun terdaftar. Silakan Sign Up dulu.');
                window.location.href = 'signup.html';
                return;
            }

            if (email === savedData.email && password === savedData.password) {
                alert('Login berhasil!');
                window.location.href = 'index.html'; // arahkan ke halaman beranda
            } else {
                alert('Email atau password salah. Coba lagi.');
            }
        });
    </script>
